🔀 ส่วนที่ 2: Workflow

🌐 Step 1: รับข้อมูลเข้า (Data Entry)

ข้อมูลเข้ามาได้ 2 ช่องทาง:

  • ช่องทาง A — WebApp: สมาชิกกรอกฟอร์ม CEO Contract → กดบันทึก → ข้อมูลส่งผ่าน Apps Script → ลง Google Sheet
  • ช่องทาง B — Google Sheet: Mentor/Service Leader เปิด Sheet แก้ไขข้อมูลโดยตรง → ข้อมูลเปลี่ยนทันที

💾 Step 2: จัดเก็บข้อมูล (Data Storage)

Google Sheets เป็น Database หลัก:

  • Sheet: CEO_Contract — เก็บข้อมูลสัญญาของสมาชิกทุกคน
  • Sheet: OKRs_KPIs — เก็บ OKR และ KPI ของแต่ละคน
  • Sheet: Task_Tracker — เก็บ Task ที่มอบหมาย
  • Sheet: Approval_Log — เก็บประวัติการอนุมัติ

🔄 Step 3: ประมวลผล & Sync (Processing)

Google Apps Script ทำหน้าที่:

  • doPost(): รับข้อมูลจาก WebApp → เขียนลง Sheet → ตอบกลับ WebApp
  • doGet(): WebApp ขอข้อมูล → อ่านจาก Sheet → ส่ง JSON กลับ
  • onEdit(): Trigger เมื่อแก้ไข Sheet → อัปเดตข้อมูลที่เกี่ยวข้อง → ส่งแจ้งเตือน
  • Time-driven: รันอัตโนมัติทุกวัน → สร้างรายงานสรุป → ส่งให้ CEO

📊 Step 4: รายงาน & อนุมัติ (Report & Approval)

ผลลัพธ์ที่ผลิตได้:

  • Dashboard รายบุคคล: แสดง CEO Contract, OKR, KPI, สถานะอนุมัติ
  • Dashboard ภาพรวม BU: สรุปทั้ง BU, Progress OKR, ทีมที่ค้างอนุมัติ
  • รายงานอนุมัติ: Member → Mentor → Service Leader (3 ระดับ)
  • แจ้งเตือน: Email/Line เมื่อมีการแก้ไข, รออนุมัติ, ครบกำหนด

📥 doPost() — รับข้อมูลจาก WebApp

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    
    if (data.action === 'save_contract') {
      const sheet = ss.getSheetByName('CEO_Contract');
      sheet.appendRow([
        data.id, data.name, data.bu, data.team, data.role,
        data.purpose, data.vision, data.accountability_json,
        'Pending', new Date(), new Date()
      ]);
      return response({ status: 'ok', message: 'บันทึกสำเร็จ' });
    }
    
    if (data.action === 'save_okr') {
      const sheet = ss.getSheetByName('OKRs');
      sheet.appendRow([
        data.id, data.member_id, data.level, data.objective,
        data.key_results, data.target, data.actual,
        'On Track', data.period
      ]);
      return response({ status: 'ok', message: 'บันทึก OKR สำเร็จ' });
    }
    
    if (data.action === 'update_approval') {
      const sheet = ss.getSheetByName('Approval_Log');
      sheet.appendRow([
        data.id, data.contract_id, data.approver_role,
        data.approver_name, data.status, new Date(), data.comment
      ]);
      
      // อัปเดตสถานะใน CEO_Contract
      const contractSheet = ss.getSheetByName('CEO_Contract');
      const data = contractSheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (data[i][0] === data.contract_id) {
          contractSheet.getRange(i + 1, 9).setValue(data.status);
          break;
        }
      }
      return response({ status: 'ok', message: 'อัปเดตการอนุมัติสำเร็จ' });
    }
    
  } catch (err) {
    return response({ status: 'error', message: err.toString() });
  }
}

function response(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

📤 doGet() — ส่งข้อมูลให้ WebApp / Dashboard

function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const action = e.parameter.action;
    
    if (action === 'get_contracts') {
      const sheet = ss.getSheetByName('CEO_Contract');
      const data = sheet.getDataRange().getValues();
      const headers = data[0];
      const rows = data.slice(1).map(r => {
        let obj = {};
        headers.forEach((h, i) => obj[h] = r[i]);
        return obj;
      });
      return response({ status: 'ok', data: rows });
    }
    
    if (action === 'get_dashboard') {
      const bu = e.parameter.bu;
      const contracts = ss.getSheetByName('CEO_Contract').getDataRange().getValues();
      const okrs = ss.getSheetByName('OKRs').getDataRange().getValues();
      
      // สรุปข้อมูล
      let summary = {
        total_members: contracts.length - 1,
        approved: 0,
        pending: 0,
        bu_progress: {}
      };
      
      for (let i = 1; i < contracts.length; i++) {
        if (contracts[i][8] === 'Approved') summary.approved++;
        else summary.pending++;
        
        let bu = contracts[i][2];
        if (!summary.bu_progress[bu]) summary.bu_progress[bu] = { total: 0, approved: 0 };
        summary.bu_progress[bu].total++;
        if (contracts[i][8] === 'Approved') summary.bu_progress[bu].approved++;
      }
      
      return response({ status: 'ok', data: summary });
    }
    
    if (action === 'get_member') {
      const memberId = e.parameter.id;
      const sheet = ss.getSheetByName('CEO_Contract');
      const data = sheet.getDataRange().getValues();
      
      for (let i = 1; i < data.length; i++) {
        if (data[i][0] === memberId) {
          let member = {};
          data[0].forEach((h, j) => member[h] = data[i][j]);
          return response({ status: 'ok', data: member });
        }
      }
      return response({ status: 'error', message: 'ไม่พบสมาชิก' });
    }
    
  } catch (err) {
    return response({ status: 'error', message: err.toString() });
  }
}

🔄 onEdit() — Trigger เมื่อแก้ไข Sheet

function onEdit(e) {
  const range = e.range;
  const sheet = range.getSheet();
  const sheetName = sheet.getName();
  
  // เมื่อมีการแก้ไขใน CEO_Contract Sheet
  if (sheetName === 'CEO_Contract') {
    const col = range.getColumn();
    const row = range.getRow();
    
    // ถ้ามีการแก้ไขสถานะ (Column 9 = สถานะ)
    if (col === 9 && row > 1) {
      const newStatus = range.getValue();
      const memberId = sheet.getRange(row, 1).getValue();
      const memberName = sheet.getRange(row, 2).getValue();
      
      // บันทึกใน Approval_Log
      const logSheet = SpreadsheetApp.getActiveSpreadsheet()
        .getSheetByName('Approval_Log');
      logSheet.appendRow([
        'LOG-' + Date.now(), memberId, 'System', 'Auto',
        'Status changed to: ' + newStatus, new Date(), ''
      ]);
      
      // ส่งแจ้งเตือน (ถ้ามี)
      sendNotification(memberName, newStatus);
    }
  }
}

🔒 ส่วนที่ 6: ความปลอดภัย

มาตรการป้องกัน

ความเสี่ยงวิธีป้องกัน
ข้อมูลถูกแก้ไขโดยไม่มีสิทธิ์ Google Sheets ตั้งค่า Share: View-only สำหรับสมาชิก, Edit เฉพาะ Mentor/Service Leader
ข้อมูลหาย/ถูก ลบ Google Sheets มี Version History กู้คืนได้, สร้าง Backup Sheet ทุกวันด้วย Time-driven Trigger
API ถูกเรียกโดยไม่มีสิทธิ์ Apps Script ตรวจสอบ parameter "secret_key" ก่อนประมวลผล
ข้อมูลซ้ำ/ขัดแย้งกัน ใช้ Member_ID เป็น Primary Key, ตรวจสอบก่อนบันทึก, onEdit() ป้องกันการแก้ไขซ้อน
